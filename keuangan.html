const allTx = $('GET FILTER LOG').all().map(i => i.json); const allExp = $('GET FILTER EXPENSE').all().map(i => i.json); const filter = $('Webhook').item.json.body; const startDate = filter.startDate; const endDate = filter.endDate; const targetCabang = filter.cabang || 'ALL'; let report = { omset: 0, bagiOwner: 0, expense: 0, laba: 0, terapis: {}, rawTransactions: [] }; allTx.forEach(tx => { // Pastikan tidak menghitung data yang masih HOLD const isHold = (tx.METODE_BAYAR || '').toUpperCase() === 'HOLD'; if (tx.TANGGAL >= startDate && tx.TANGGAL <= endDate && !isHold) { if (targetCabang === 'ALL' || tx.CABANG_ID === targetCabang) { const tagihan = Number(tx.TAGIHAN) || 0; const bagiOwner = Number(tx.BAGI_OWNER) || 0; const bagiTerapis = Number(tx.BAGI_TERAPIS) || 0; report.omset += tagihan; report.bagiOwner += bagiOwner; const rawTerapis = tx.TERAPIS ? String(tx.TERAPIS).trim().toUpperCase() : "TANPA NAMA"; const cid = tx.CABANG_ID || "UNKNOWN"; const key = cid + "_" + rawTerapis; if (!report.terapis[key]) report.terapis[key] = { cabang: cid, nama: rawTerapis, totalPasien: 0, totalKomisi: 0 }; report.terapis[key].totalPasien += 1; report.terapis[key].totalKomisi += bagiTerapis; } } }); allExp.forEach(ex => { if (ex.TANGGAL >= startDate && ex.TANGGAL <= endDate) { if (targetCabang === 'ALL' || ex.CABANG_ID === targetCabang) { report.expense += (Number(ex.NOMINAL) || 0); } } }); report.laba = report.bagiOwner - report.expense; report.daftarTerapis = Object.values(report.terapis).sort((a,b)=>b.totalKomisi - a.totalKomisi); return [{ json: { status: "success", data: report } }];
